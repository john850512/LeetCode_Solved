# 2. Add Two Numbers
- https://leetcode.com/problems/add-two-numbers/
#### Approach 1: Brute Force
分析:
- Time Complexity: O(max(M, N))
- Space Complexity: 最多是O(max(M, N))+1(進位時)
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* head = new ListNode(-1);
        ListNode* ptr = head;
        
        while(l1 != NULL && l2 != NULL){
            ListNode* new_ptr = new ListNode(l1->val + l2->val);
            ptr->next = new_ptr;
            ptr = ptr->next;
            l1 = l1->next;
            l2 = l2->next;
        }
        while(l1 != NULL){
            ListNode* new_ptr = new ListNode(l1->val);
            ptr->next = new_ptr;
            ptr = ptr->next;
            l1 = l1->next;
        }
        while(l2 != NULL){
            ListNode* new_ptr = new ListNode(l2->val);
            ptr->next = new_ptr;
            ptr = ptr->next;
            l2 = l2->next;
        }
        
        ptr = head->next;
        while(ptr != NULL){
            if(ptr->val / 10){
                if(ptr->next == NULL){
                    ptr->next = new ListNode(ptr->val / 10);
                    ptr->val %= 10;
                }
                else{
                    ptr->next->val += ptr->val / 10;
                    ptr->val %= 10;
                }
            }
            ptr = ptr->next;
        }
        return head->next;
    }
};
```

#### 2.
不要另外創一個Linked list存結果，直接把結果加在某一邊
分析:
- Time Complexity: O(max(M, N))
- Space Complexity: 最多是O(max(M, N))，雖然都一樣，但比上面少了整整一倍(上面其實是2*O(max(M, N)))
``` c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* head = l1;
        
        while(l1 != NULL && l2 != NULL){
            printf("%d %d \n", l1->val, l2->val);
            l1->val += l2->val;
            if(l1->next != NULL && l2->next != NULL){
                l1 = l1->next;
                l2 = l2->next;
            }
            else{
                break;
            }
        }
        while(l2->next != NULL){
            l1->next =  new ListNode(l2->next->val);
            l1 = l1->next;
            l2 = l2->next;
        }
        l1 = head;
        while(l1 != NULL){
            if(l1->val / 10){
                if(l1->next == NULL){
                    l1->next = new ListNode(l1->val / 10);
                }
                else{
                    l1->next->val += l1->val / 10;
                }
                l1->val %= 10;
            }
            l1 = l1->next;
        }
        return head;
    }
};
```
