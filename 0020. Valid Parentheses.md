# 20. Valid Parentheses
#### Approach 1. stack
分析:
- Time complexity: O(N), N is number of char
- Space complexity: O(N)
```c++
class Solution {
public:
    bool isValid(string s) {
        stack<char> t;
        for(auto i: s){
            if(i == '(' || i == '['  || i == '{' ) t.push(i);
            else{
                if(t.empty()) return false;
                switch(i){
                    case ')':
                        if(t.top() != '(') return false;
                        t.pop();
                        break;
                    case ']':
                        if(t.top() != '[') return false;
                        else t.pop();
                        break;
                    case '}':
                        if(t.top() != '{') return false;
                        else t.pop();
                        break;
                }
            }
        }
        if(t.empty()) return true;
        else return false;
    }
};
```
